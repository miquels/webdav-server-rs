# ssl certificate 
openssl req -x509 -nodes -new -sha256 -days 1024 -newkey rsa:2048 -keyout RootCA.key -out RootCA.pem -subj "/C=US/CN=Example-Root-CA"
openssl x509 -outform pem -in RootCA.pem -out RootCA.crt
echo -n "authorityKeyIdentifier=keyid,issuer
basicConstraints=CA:FALSE
keyUsage = digitalSignature, nonRepudiation, keyEncipherment, dataEncipherment
subjectAltName = @alt_names
[alt_names]
DNS.1 = localhost
" > domains.ext
openssl req -new -nodes -newkey rsa:2048 -keyout localhost.key -out localhost.csr -subj "/C=US/ST=YourState/L=YourCity/O=Example-Certificates/CN=Example-localhost"
openssl x509 -req -sha256 -days 365 -in localhost.csr -CA RootCA.pem -CAkey RootCA.key -CAcreateserial -extfile domains.ext -out localhost.crt
# mac
sudo security add-trusted-cert -d -r trustRoot -k /Library/Keychains/System.keychain RootCA.crt 
# windows
RootCA.crt import
In the search box, begin typing mmc.exe, right-click the mmc.exe entry in the search results and select Run as Administrator.
Select File > Add/Remove Snap-in.
Select Certificates and click Add.
In the Certificates snap-in dialog, select Computer account and complete the wizard.
In the MMC console, expand Certificates.
Right-click Trusted Root Certificates and select All Tasks > Import.
